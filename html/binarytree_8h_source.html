541 TreeNode<T> * BinaryTree<T>::reconstructTree(vector<T> * inorderList, vector<T> * preorderList)
542 {
543  typename vector<T>::iterator itr = std::find(inorderList->begin(),
544  inorderList->end(),
545  preorderList->at(0));
546  TreeNode<T> * currentRoot = new TreeNode<T>(*itr);
547  if(itr != inorderList->begin())
548  {
549  vector<T> * leftSubTree = new vector<T>(inorderList->begin(), itr);
550  vector<T> * leftSubTreePre = new vector<T>(preorderList->begin() + 1, preorderList->begin() + 1 + leftSubTree->size());
551  TreeNode<T> * leftSubTreeRoot = reconstructTree(leftSubTree, leftSubTreePre);
552  currentRoot->setLeft(leftSubTreeRoot);
553  leftSubTreeRoot->setParent(currentRoot);
554  }
555  if(itr + 1 != inorderList->end())
556  {
557  vector<T> * rightSubTree = new vector<T>(itr + 1, inorderList->end());
558  vector<T> * rightSubTreePre = new vector<T>(preorderList->begin() + inorderList->size() - rightSubTree->size(),
559  preorderList->begin() + inorderList->size());
560  TreeNode<T> * rightSubTreeRoot = reconstructTree(rightSubTree, rightSubTreePre);
561  currentRoot->setRight(rightSubTreeRoot);
562  rightSubTreeRoot->setParent(currentRoot);
563  }
564  return currentRoot;
565 }

763  bfsQueue->push(root);

767  {
768  TreeNode<T> * currNode = bfsQueue->front();
769  bfsQueue->pop();
770  nodeArray->push_back(currNode);
771  if(currNode == sentinel)
772  {
773  levels->push_back(currentLevel);
774  currentLevel = new LinkedList<TreeNode<T> * >;
775  continue;
776  }
777  if(currNode->getLeft())
778  {
779  bfsQueue->push(currNode->getLeft());
780  }
781  if(currNode->getRight())
782  {
783  bfsQueue->push(currNode->getRight());
784  }
785  if(bfsQueue->front() == sentinel)
786  {
787  bfsQueue->push(sentinel);
788  }
789  currentLevel->append(new Node<TreeNode<T> * >(currNode));
790  }

941 template<class T>
942 TreeNode<T> * BinaryTree<T>::findFirstCommonAncestor(TreeNode<T> * p, TreeNode<T> * q, TreeNode<T> * root)
943 {
944  // if root itself is p or q just return it
945  if(*root == *p || *root == *q)
946  {
947  return root;
948  }
949  TreeNode<T> * foundLeft = NULL, * foundRight = NULL;
950  if(root->getLeft())
951  {
952  // if there is a left child recursively find
953  foundLeft = findFirstCommonAncestor(p, q, root->getLeft());
954  if(foundLeft && *foundLeft != *p && *foundLeft != *q)
955  {
956  // the recursive call returned the first common ancestor
957  // just return it
958  return foundLeft;
959  }
960  }
961  if(root->getRight())
962  {
963  // if there is a right child recursively find
964  foundRight = findFirstCommonAncestor(p, q, root->getRight());
965  if(foundRight && *foundRight != *p && *foundRight != *q)
966  {
967  // the recursive call returned the first common ancestor
968  // just return it
969  return foundRight;
970  }
971  }
972  // either p and q are on either sides or only one of them is present in one of the sub trees
973  // or none of them is found
974  if(!foundLeft && !foundRight)
975  {
976  return NULL;
977  }
978  if(!foundLeft) return foundRight;
979  else if(!foundRight) return foundLeft;
980 
981  // by now we know that both of them are not null and both of them are either p or q
982  // by assumption that the nodes are unique, we can just return the root here
983  // since root is the common ancestor
984  return root;
985 }

1020  }
